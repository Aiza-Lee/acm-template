\textbf{卡特兰数}

\textbf{定义}
$H_0 = 1, H_n = \sum_{i=0}^{n-1} H_i H_{n-1-i}$

\textbf{通项公式}
\[ H_n = \frac{\binom{2n}{n}}{n+1} = \binom{2n}{n} - \binom{2n}{n-1} \]

\textbf{常见应用}
\begin{itemize}
    \item $n$ 个节点的二叉树数量
    \item $n$ 对括号的合法序列数量
    \item $n+2$ 边形的三角剖分数量
    \item $n \times n$ 网格从 $(0,0)$ 到 $(n,n)$ 不越过对角线的路径数量
\end{itemize}
