\textbf{广义二项式定理}

广义二项式定理将二项式定理推广到实数指数 $\alpha$ 的情况。

\[
(x+y)^\alpha = \sum_{k=0}^\infty \binom{\alpha}{k} x^k y^{\alpha-k} \quad (|x| < |y|)
\]

其中广义二项式系数定义为：
\[
\binom{\alpha}{k} = \frac{\alpha (\alpha - 1) \cdots (\alpha - k + 1)}{k!}
\]

\textbf{特殊形式 (牛顿二项式定理)}
当 $y=1$ 时：
\[
(1+x)^\alpha = \sum_{k=0}^\infty \binom{\alpha}{k} x^k \quad (|x| < 1)
\]

\textbf{常用展开式}
\begin{enumerate}
    \item 负整数指数 $\alpha = -n$：
    \[
    \binom{-n}{k} = (-1)^k \binom{n+k-1}{k}
    \]
    \[
    (1-x)^{-n} = \sum_{k=0}^\infty \binom{n+k-1}{k} x^k = \sum_{k=0}^\infty \binom{n+k-1}{n-1} x^k
    \]
    \item 特例 $n=1$ (几何级数)：
    \[
    \frac{1}{1-x} = \sum_{k=0}^\infty x^k
    \]
    \item 半整数指数 (如 $\sqrt{1+x}$)，利用卡特兰数相关系数：
    \[
    \binom{1/2}{k} = \frac{(-1)^{k-1}}{k \cdot 2^{2k-1}} \binom{2k-2}{k-1}
    \]
\end{enumerate}
