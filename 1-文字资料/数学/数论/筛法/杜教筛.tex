\textbf{杜教筛}

\textbf{目标}
求
\[
S(n)=\sum_{i=1}^{n}f(i),
\]
其中 $f$ 常为积性函数。

\textbf{核心构造}
选一个卷积伙伴 $g$，令
\[
h=f*g.
\]
则
\[
\sum_{i=1}^{n}h(i)=\sum_{d=1}^{n}g(d)\,S\!\left(\left\lfloor\frac{n}{d}\right\rfloor\right).
\]
移项得
\[
g(1)S(n)=\sum_{i=1}^{n}h(i)-\sum_{d=2}^{n}g(d)\,S\!\left(\left\lfloor\frac{n}{d}\right\rfloor\right).
\]

\textbf{可用条件}
\begin{itemize}
\item $\sum_{i=1}^{n}h(i)$ 易求。
\item $\sum_{i=1}^{n}g(i)$ 易求或可预处理。
\item 对 $\left\lfloor n/d\right\rfloor$ 做整除分块，并配合记忆化。
\end{itemize}

\textbf{复杂度}
基础版常记为
\[
O(n^{3/4}),
\]
若预处理一段前缀和，常可做到
\[
O(n^{2/3}).
\]

\textbf{常见选择}
\begin{itemize}
\item 求 $\sum_{i=1}^{n}\varphi(i)$：可取 $g=1$，则 $h=\mathrm{id}$。
\item 求 $\sum_{i=1}^{n}\mu(i)$：可取 $g=1$，则 $h=\varepsilon$。
\end{itemize}
