\textbf{中国剩余定理}

\textbf{标准形}
\[
x\equiv a_i \pmod{m_i}\qquad (1\le i\le n)
\]

\textbf{CRT 适用条件}
\[
\gcd(m_i,m_j)=1\qquad (i\ne j)
\]
此时模
\[
M=\prod_{i=1}^n m_i
\]
下解唯一。

\textbf{直接公式}
设
\[
M_i=\frac{M}{m_i},\qquad t_i\equiv M_i^{-1}\pmod{m_i},
\]
则
\[
x\equiv \sum_{i=1}^n a_iM_it_i \pmod M.
\]

\textbf{扩展 CRT}
两式
\[
x\equiv a_1\pmod{m_1},\qquad x\equiv a_2\pmod{m_2}
\]
可化为
\[
m_1k-m_2\ell=a_2-a_1.
\]
设 $g=\gcd(m_1,m_2)$，则有解当且仅当
\[
g\mid (a_2-a_1).
\]
有解时可合并为
\[
x\equiv a \pmod{\mathrm{lcm}(m_1,m_2)}.
\]
多式情形按顺序两两合并。

\textbf{快记}
\begin{itemize}
\item CRT：模数两两互质，直接套总公式。
\item ExCRT：模数不互质，先判 $(a_2-a_1)$ 是否被 $\gcd$ 整除，再并到 $\mathrm{lcm}$。
\end{itemize}
