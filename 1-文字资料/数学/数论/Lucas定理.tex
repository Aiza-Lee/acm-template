\textbf{Lucas 定理}

\textbf{适用条件}
$p$ 为素数。

\textbf{标准式}
对非负整数 $n,m$，
\[
\binom{n}{m}\equiv
\binom{\lfloor n/p\rfloor}{\lfloor m/p\rfloor}
\binom{n\bmod p}{m\bmod p}
\pmod p.
\]

\textbf{$p$ 进制形式}
若
\[
n=\sum_i n_ip^i,\qquad m=\sum_i m_ip^i,
\]
则
\[
\binom{n}{m}\equiv \prod_i \binom{n_i}{m_i}\pmod p.
\]
若某位 $m_i>n_i$，该项直接为 $0$。

\textbf{扩展 Lucas}
要求
\[
\binom{n}{m}\pmod M,\qquad M=\prod_i p_i^{c_i},
\]
做法是：
\begin{itemize}
\item 分别计算 $\binom{n}{m}\pmod{p_i^{c_i}}$；
\item 再用 ExCRT 合并。
\end{itemize}

\textbf{模 $p^k$ 的核心拆法}
\[
\binom{n}{m}=\frac{n!}{m!(n-m)!}
\]
但分母可能含 $p$，不能直接求逆。
常写成
\[
n!=p^{v_p(n!)}\cdot f(n),\qquad
v_p(n!)=\sum_{j\ge 1}\left\lfloor\frac{n}{p^j}\right\rfloor,
\]
其中 $f(n)$ 表示去掉所有 $p$ 因子的部分，再利用模 $p^k$ 下的周期性递归计算。

\textbf{快记}
普通 Lucas 处理“模素数的大组合数”，ExLucas = “拆质因数幂 + ExCRT”。
