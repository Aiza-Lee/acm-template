\textbf{基本构造}

\textbf{1. 空类 (Empty Class)} \\
含有一个大小为 $0$ 的对象。
\[ \mathcal E \]
OGF：$E(x)=1$

\textbf{2. 原子类 (Atomic Class)} \\
含有一个大小为 $1$ 的对象。
\[ \mathcal Z \]
OGF：$Z(x)=x$

\textbf{3. 不相交并 (Disjoint Union)} \\
集合的并。
\[ \mathcal A + \mathcal B \]
OGF：$A(x) + B(x)$ \quad 系数：$c_n = a_n + b_n$

\textbf{4. 笛卡尔积 (Cartesian Product)} \\
\[ \mathcal A \times \mathcal B \]
OGF：$A(x)B(x)$ \quad 系数：$c_n = \sum_{i+j=n}a_ib_j$

\textbf{复合构造}

\textbf{5. 序列 (Sequence)} \\
$\mathcal A$ 中对象构成的序列。
\[ \text{SEQ}(\mathcal A) = \mathcal E + (\mathcal A \times \mathcal A) + (\mathcal A \times \mathcal A \times \mathcal A) + \cdots \]
OGF：
\[ \sum_{k \ge 0} A(x)^k = \frac{1}{1-A(x)} \]

\textbf{6. 多重集 (Multiset)} \\
$\mathcal A$ 中对象构成的无序多重集。
\[ \text{MSET}(\mathcal A) =\text{SEQ}(\mathcal A)/\mathfrak S =  \prod_{\alpha \in \mathcal A} \text{SEQ}(\{ \alpha \}) \]
OGF：
\[ \text{Euler}(A(x)) = \prod_{n \ge 1}(1-x^n)^{-a_n} = \exp\left( \sum_{k\ge 1}\frac{A(x^k)}{k} \right) \]

\textbf{7. 幂集 (Power Set)} \\
$\mathcal A$ 的子集。
\[ \text{PSET}(\mathcal A) = \prod_{\alpha \in \mathcal A}(\mathcal E + \{\alpha\}) \]
OGF：
\[ \prod_{n\ge 1}(1+x^n)^{a_n} = \exp\left(\sum_{k \ge 1}(-1)^{k + 1}\frac{A(x^k)}{k}\right) \]

\textbf{8. 循环 (Cycle)} \\
$\mathcal A$ 中对象构成的循环排列。
\[ \text{CYC}(\mathcal{A}) = (\text{SEQ}(\mathcal{A}) \setminus \mathcal{E}) / \mathfrak{C} \]
OGF：
\[ \sum_{k \ge 1} \frac{\varphi(k)}{k} \ln \frac{1}{1 - A(x^k)} \]
